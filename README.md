## Usage

```jsx
import React from "react"
import ReactDOM from "react-dom"
import ReactImageMagician from "react-image-magician"

ReactDOM.render(
    <ReactImageMagician mode="base64">
        <img src="YOUR_IMG_URL"/>
    </ReactImageMagician>,
    document.getElementById('root')
)
```
